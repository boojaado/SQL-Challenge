titles
-
ID Serial PK
title_id varchar(10)
title varchar(50)
last_updated timestamp

departments
-
ID Serial PK
dept_no varchar(10)
dept_name varchar(50)
last_updated timestamp

employees
-
ID Serial PK
emp_no int
emp_title_id varchar(10) FK >- titles.title_id
birth_date date
first_name varchar(100)
last_name varchar(100)
hire_date date
last_updated timestamp

dept_manager
-
ID Serial PK
dept_no varchar(10) FK >- departments.dept_no
emp_no int FK >- employees.emp_no
last_updated timestamp

salaries
-
ID Serial PK
emp_no int FK >- employees.emp_no
salary int
last_updated timestamp

dept_emp
-
ID Serial PK
